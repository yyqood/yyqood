this指向问题

谁调用函数，函数的this就指向谁，否则指向全局变量window

1.普通函数调用：this指向window
function f(){
console.log(this);
}
f(); //等价于 window.f();
2.构造函数调用：this指向实例对象
function Student(age,name){
this.age = age;
this.name = name;
console.log(this);
}
var s1 = new Student(12,”s1”);
var s2 = new Student(13,”s2”);
3.作为方法被调用：this指向该方法所属的对象
var obj = {
f:function(){
console.log(this);
}
}
obj.f(); //obj

更正this指向

1.call()方法
var Person = {
name:”qyy”,
age:100
}
function f1(x,y){
console.log(x,”+”,y);
console.log(this);
}
f1(“ss”,88);
//此时this指向window
function f2(x,y){
console.log(x,”+”,y);
console.log(this);
console.log(this.name);
console.log(this.age);
}
f2.call(Person,”ss”,88); //第一个参数为指定的this指向对象
